%================= CODE =================
\section{Cài đặt}

\begin{frame}[fragile]{Code DFS (C++)}
\begin{tcolorbox}
\begin{lstlisting}
#include <bits/stdc++.h>
using namespace std;

vector<vector<int>> adj;
vector<bool> visited;

void dfs(int u) {
    visited[u] = true;

    // Xu ly dinh u
    cout << u << " ";

    for (int v : adj[u]) {
        if (!visited[v]) {
            dfs(v);
        }
    }
}

/*
Phan tich:
- Thoi gian: O(V + E)
- Moi dinh duyet 1 lan
- Su dung de quy don gian
*/
\end{lstlisting}
\end{tcolorbox}
\end{frame}
