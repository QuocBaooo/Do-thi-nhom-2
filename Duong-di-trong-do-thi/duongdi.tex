\section{Dijkstra}
\begin{frame}{Thuật toán Dijkstra}

\begin{columns}

\column{0.5\textwidth}
\begin{tcolorbox}[title=\faWalking \ Ý tưởng]
Tìm đường từ một điểm đến tất cả các điểm còn lại.
\begin{itemize}
    \item[\faCheck] Chỉ dùng cho cạnh \textbf{dương}.
    \item[\faStar] Luôn chọn điểm \textbf{gần nhất} để đi tiếp.
\end{itemize}
\end{tcolorbox}

\vspace{0.2cm}

\begin{tcolorbox}[colframe=main2, title=\faSync \ Quy trình]
\begin{enumerate}
    \item Chọn điểm $U$ gần nhất.
    \item Xem các điểm $V$ quanh nó.
    \item Nếu đi qua $U$ mà \textbf{ngắn hơn} đường cũ $\to$ Cập nhật.
\end{enumerate}
\end{tcolorbox}

\column{0.5\textwidth}
\begin{center}
\begin{tikzpicture}[node distance={15mm}, thick, main/.style = {draw, circle, fill=main1!20}] 
\node[main] (1) {S}; 
\node[main] (2) [above right of=1] {A}; 
\node[main] (3) [below right of=1] {B}; 
\node[main] (4) [right of=2] {C};

\draw (1) -- node[midway, sloped, above] {7} (2); 
\draw (1) -- node[midway, sloped, below] {2} (3); 
\draw (3) -- node[midway, sloped, above] {3} (2);
\draw (2) -- node[midway, sloped, above] {1} (4);
\end{tikzpicture}
\end{center}

\begin{tcolorbox}
\small \faLightbulb \textbf{Ví dụ:} Đi từ S đến A trực tiếp mất 7, nhưng qua B chỉ mất $2+3=5$. Ta chọn đường qua B.
\end{tcolorbox}

\end{columns}
\end{frame}

\section{Floyd-Warshall}
\begin{frame}{Thuật toán Floyd-Warshall}

\begin{columns}

\column{0.5\textwidth}
\begin{tcolorbox}[title=\faMap \ Ý tưởng]
Tìm mọi con đường ngắn nhất giữa \textbf{mọi cặp điểm}.
\begin{itemize}
    \item[\faCheck] Dùng được cho cạnh \textbf{âm}.
    \item[\faRedo] Thử chèn từng điểm trung gian vào giữa.
\end{itemize}
\end{tcolorbox}

\vspace{0.2cm}

\begin{tcolorbox}[colframe=main2, title=\faQuestion \ So sánh]
Với mọi cặp điểm (A, B):
\begin{itemize}
    \item[\faArrowRight] Cách 1: $A \to B$
    \item[\faArrowRight] Cách 2: $A \to K \to B$
\end{itemize}
\textbf{Cái nào ngắn hơn thì lấy!}
\end{tcolorbox}

\column{0.5\textwidth}
\begin{center}
\begin{tikzpicture}[scale=0.8]
\node[circle,draw,fill=main2!20] (I) at (0,0) {i};
\node[circle,draw,fill=main2!20] (J) at (3,0) {j};
\node[circle,draw,fill=main1!40] (K) at (1.5,1.5) {k};

\draw[->, dashed] (I) -- (J) node[midway, below] {Đường thẳng};
\draw[->, thick, main1] (I) -- (K);
\draw[->, thick, main1] (K) -- (J);
\node at (1.5, -0.8) {\small \textit{Thử đi vòng qua K}};
\end{tikzpicture}
\end{center}

\begin{tcolorbox}
\textbf{\faInfoCircle Đặc điểm}
\begin{itemize}
    \item Rất dễ cài đặt (3 vòng lặp).
    \item Chạy chậm hơn Dijkstra trên đồ thị lớn.
\end{itemize}
\end{tcolorbox}

\end{columns}
\end{frame}

\section{So sánh}
\begin{frame}{Nên chọn thuật toán nào?}

\begin{columns}

\column{0.5\textwidth}
\begin{tcolorbox}[title=\faBolt \ Dùng Dijkstra khi...]
\begin{itemize}
    \item[\faMapMarker*] Chỉ cần tìm từ 1 điểm xuất phát.
    \item[\faHashtag] Đồ thị rất nhiều đỉnh và cạnh (như bản đồ thực).
    \item[\faBan] Không có đường đi trọng số âm.
\end{itemize}
\end{tcolorbox}

\column{0.5\textwidth}
\begin{tcolorbox}[title=\faLayerGroup \ Dùng Floyd khi...]
\begin{itemize}
    \item[\faObjectGroup] Cần biết mọi đường đi giữa mọi cặp.
    \item[\faExpandArrowsAlt] Đồ thị nhỏ (ít hơn 500 điểm).
    \item[\faCheckDouble] Có xuất hiện trọng số âm.
\end{itemize}
\end{tcolorbox}

\end{columns}
